    Implement Levenshtein distance inline (no external dependency). Add above `parseConfig`:

    ```typescript
    function levenshtein(a: string, b: string): number {
      const m = a.length, n = b.length;
      const dp: number[][] = Array.from({ length: m + 1 }, (_, i) =>
        Array.from({ length: n + 1 }, (_, j) => (i === 0 ? j : j === 0 ? i : 0))
      );
      for (let i = 1; i <= m; i++) {
        for (let j = 1; j <= n; j++) {
          dp[i][j] = a[i - 1] === b[j - 1]
            ? dp[i - 1][j - 1]
            : 1 + Math.min(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1]);
        }
      }
      return dp[m][n];
    }

    function suggestClosest(input: string, candidates: string[]): string | null {
      if (!input || candidates.length === 0) return null;
      const threshold = Math.max(2, Math.floor(input.length * 0.4));
      // Case-insensitive comparison
      const lower = input.toLowerCase();
      let best: string | null = null;
      let bestDist = Infinity;
      for (const c of candidates) {
        const dist = levenshtein(lower, c.toLowerCase());
        if (dist <= threshold && dist < bestDist) {
          bestDist = dist;
          best = c;
        }
      }
      return best;
    }
    ```
