    function formatZodPath(path: (string | number)[]): string {
      return path.reduce<string>((acc, segment, i) => {
        if (typeof segment === "number") return `${acc}[${segment}]`;
        return i === 0 ? segment : `${acc}.${segment}`;
      }, "");
    }
