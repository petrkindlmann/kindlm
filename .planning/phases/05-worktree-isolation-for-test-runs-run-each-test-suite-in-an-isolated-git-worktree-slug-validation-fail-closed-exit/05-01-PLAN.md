```typescript
export class WorktreeError extends Error { ... }
export class WorktreeHasChangesError extends WorktreeError {
  constructor(public readonly worktreePath: string, public readonly files: number, public readonly commits: number) { ... }
}

export function validateWorktreeSlug(slug: string): void
// Rules:
//   1. Length: 1–64 chars. Throw WorktreeError("slug exceeds 64 characters") if over.
//   2. Charset: /^[a-zA-Z0-9._-]+$/ only. Throw WorktreeError("slug contains invalid characters: only a-z A-Z 0-9 . _ - allowed").
//   3. Reserved segments: split on "/" (there won't be slashes after rule 2, but also check the slug itself).
//      If slug === "." or slug === ".." throw WorktreeError("slug uses reserved path segment").
//      Also reject if any ".." appears as a complete segment if slug were used as a path component (slug.split("/").includes("..")).

export async function createWorktree(
  slug: string,
  repoRoot?: string,  // defaults to process.cwd()
): Promise<{ path: string; branch: string; cleanup(): Promise<void> }>
// Steps:
//   1. validateWorktreeSlug(slug) — throws early if invalid.
//   2. worktreePath = path.join(repoRoot ?? process.cwd(), ".kindlm", "worktrees", slug)
//   3. branch = `kindlm-isolate/${slug}`
//   4. Run: git worktree add --detach {worktreePath}
//      (detached HEAD — no branch creation, avoids branch name conflicts)
//   5. Return { path: worktreePath, branch: "HEAD", cleanup }
//   cleanup():
//     - Call removeWorktree(worktreePath, false)
//     - If WorktreeHasChangesError: warn to console.warn, print path, swallow — do NOT rethrow

export async function removeWorktree(worktreePath: string, force?: boolean): Promise<void>
// If force !== true:
//   - Call countWorktreeChanges(worktreePath)
//   - If result is null (cannot determine) → treat as fail-closed: throw WorktreeHasChangesError(worktreePath, -1, -1)
//   - If result.files > 0 or result.commits > 0 → throw WorktreeHasChangesError(worktreePath, files, commits)
// Run: git worktree remove {force ? "--force " : ""}{worktreePath}

export async function countWorktreeChanges(
  worktreePath: string,
): Promise<{ files: number; commits: number } | null>
// files: run `git -C {worktreePath} status --porcelain` — count non-empty lines
// commits: run `git -C {worktreePath} log @{u}..HEAD --oneline` — count lines
//   If the upstream lookup fails (no upstream set), commits = 0
// If any execFile call rejects with ENOENT or exits non-zero on `git status`: return null (fail-closed)
```

```typescript
function toWorktreeSlug(suite: string, runId: string): string {
  // Replace non-[a-zA-Z0-9._-] chars with "-", collapse runs of "-", max 55 chars, append "-" + runId (8 chars) = max 64
  const sanitized = suite
    .replace(/[^a-zA-Z0-9._-]/g, "-")
    .replace(/-{2,}/g, "-")
    .replace(/^-|-$/g, "")
    .slice(0, 55);
  return `${sanitized || "run"}-${runId}`;
}
```
